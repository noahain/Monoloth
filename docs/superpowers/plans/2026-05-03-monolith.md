> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Python desktop app with pywebview that lets users pick a directory and then launches OpenCode inside an embedded xterm.js terminal.

**Architecture:** Python backend exposes a JS bridge API for directory picking, PTY spawning, and input forwarding. Frontend uses xterm.js for terminal emulation. Data flows through pywebview's built-in `evaluate_js` and `pywebview.api` mechanisms.

**Tech Stack:** Python, pywebview, pywinpty, HTML/CSS/JS, xterm.js

Create `pty_manager.py`:
```python
import threading
import time


class PtyManager:
    def __init__(self):
        self._pty = None
        self._process = None
        self._write_callback = None
        self._read_thread = None
        self._running = False

    def spawn(self, command, cwd, on_output=None):
        """Spawn a PTY with the given command in the specified working directory."""
        import pywinpty
        import shutil
        import os

        if not shutil.which(command):
            raise FileNotFoundError(f"Command not found: {command}")

        self._write_callback = on_output
        self._pty = pywinpty.PTY(80, 24)

        env = os.environ.copy()
        self._process = self._pty.spawn(
            shutil.which('cmd.exe') or 'cmd.exe',
            env=env,
            cwd=cwd
        )

        # Give cmd.exe a moment to start, then send the command
        time.sleep(0.1)
        self._pty.write(f"{command}\r\n")

        self._running = True
        self._read_thread = threading.Thread(target=self._read_loop, daemon=True)
        self._read_thread.start()

    def _read_loop(self):
        """Background thread that reads PTY output and forwards it."""
        while self._running:
            try:
                data = self._pty.read()
                if data:
                    if self._write_callback:
                        self._write_callback(data)
                else:
                    time.sleep(0.01)
            except Exception:
                break
        if self._write_callback:
            self._write_callback("\r\n[opencode exited]\r\n")

    def write_input(self, data):
        """Write raw input into the PTY."""
        if self._pty and self._running:
            self._pty.write(data)

    def terminate(self):
        """Kill the PTY process and stop the read thread."""
        self._running = False
        if self._pty:
            try:
                self._pty.close()
            except Exception:
                pass
            self._pty = None
        if self._read_thread and self._read_thread.is_alive():
            self._read_thread.join(timeout=1)
```

import tkinter as tk
from tkinter import filedialog
import shutil
import threading

from pty_manager import PtyManager


class Api:
    def __init__(self):
        self._pty_manager = None
        self._window = None

    def set_window(self, window):
        self._window = window

    def pick_directory(self):
        """Open a native folder picker and return the selected path, or None if cancelled."""
        root = tk.Tk()
        root.withdraw()
        root.attributes('-topmost', True)
        path = filedialog.askdirectory(title='Choose Directory')
        root.destroy()
        if path and path.strip():
            return path.strip()
        return None

    def start_opencode(self, directory):
        """Spawn opencode in the given directory inside a PTY."""
        if not shutil.which('opencode'):
            self._write_to_terminal(
                "Error: 'opencode' command not found. "
                "Make sure it's installed and in your PATH.\r\n"
            )
            return False

        self._pty_manager = PtyManager()
        self._pty_manager.spawn(
            'opencode',
            directory,
            on_output=self._write_to_terminal
        )
        return True

    def send_input(self, data):
        """Forward terminal input to the PTY."""
        if self._pty_manager:
            self._pty_manager.write_input(data)

    def terminate(self):
        """Clean up the PTY on shutdown."""
        if self._pty_manager:
            self._pty_manager.terminate()
            self._pty_manager = None

    def _write_to_terminal(self, data):
        """Push data to the frontend terminal via JS evaluation."""
        if self._window:
            import json
            escaped = json.dumps(data)
            self._window.evaluate_js(f'window.writeToTerm({escaped})')
```

(function () {
    // --- Landing Page Logic ---
    const chooseBtn = document.getElementById('choose-dir-btn');
    if (chooseBtn) {
        chooseBtn.addEventListener('click', async () => {
            if (!window.pywebview || !window.pywebview.api) {
                alert('Python bridge not available');
                return;
            }
            try {
                const path = await window.pywebview.api.pick_directory();
                if (path) {
                    window.location.href = 'terminal.html?dir=' + encodeURIComponent(path);
                }
            } catch (err) {
                console.error('Failed to pick directory:', err);
            }
        });
    }

    // --- Terminal Page Logic ---
    const terminalContainer = document.getElementById('terminal');
    if (terminalContainer) {
        const term = new Terminal({
            theme: {
                background: '#1e1e1e',
                foreground: '#c9d1d9',
                cursor: '#c9d1d9',
                selectionBackground: '#264f78'
            },
            fontFamily: '"Cascadia Code", "Fira Code", "Consolas", monospace',
            fontSize: 14,
            cursorBlink: true
        });

        term.open(terminalContainer);
        term.focus();

        // Fit terminal to container
        function fitTerminal() {
            const cols = Math.floor(terminalContainer.clientWidth / term._core._renderService.dimensions.actualCellWidth) || 80;
            const rows = Math.floor(terminalContainer.clientHeight / term._core._renderService.dimensions.actualCellHeight) || 24;
            term.resize(cols, rows);
        }

        window.addEventListener('resize', fitTerminal);
        // Initial fit after a short delay to ensure dimensions are available
        setTimeout(fitTerminal, 100);

        // Bridge for Python to push data into the terminal
        window.writeToTerm = (data) => {
            term.write(data);
        };

        // Forward keyboard input to Python
        term.onData((data) => {
            if (window.pywebview && window.pywebview.api) {
                window.pywebview.api.send_input(data);
            }
        });

        // Start opencode in the selected directory
        const params = new URLSearchParams(window.location.search);
        const dir = params.get('dir');
        if (dir && window.pywebview && window.pywebview.api) {
            window.pywebview.api.start_opencode(dir)
                .then((success) => {
                    if (!success) {
                        term.write("\r\nFailed to start opencode.\r\n");
                    }
                })
                .catch((err) => {
                    term.write("\r\nError starting opencode: " + err + "\r\n");
                });
        } else if (!dir) {
            term.write("\r\nNo directory specified.\r\n");
        }
    }
})();
```
